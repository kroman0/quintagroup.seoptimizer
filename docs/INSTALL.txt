To install development version of the quintagroup.seoproperties package add it to buildout.cfg this way

[src-eggs]
recipe=infrae.subversion
urls =
    http://svn.quintagroup.com/products/quintagroup.seoptimizer/trunk quintagroup.seoptimizer
as_eggs = true
location = src

[instance]
zcml =
    quintagroup.seoptimizer
    quintagroup.seoptimizer-overrides
